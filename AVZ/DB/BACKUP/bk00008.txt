procedure AddBackUpData(ARoot, AName : string;Lines : TStrings);
begin
 Lines.Add('[-'+ ARoot + '\' + AName + ']');
 ExpRegKeyEx(ARoot, AName, Lines);
 Lines.Add('');
end;

var
 Lines : TStrings;
begin
 Lines := TStringList.Create;
 Lines.Add('REGEDIT4');
 Lines.Add('');
 AddBackUpData('HKEY_LOCAL_MACHINE', 'SYSTEM\CurrentControlSet\Services', Lines);
  
 if SaveBackupDataSt(Lines, 'User_drv_and_svc.reg') then
  AddToLog('$AVZ1236 $AVZ1002');
 Lines.Free;
end.
