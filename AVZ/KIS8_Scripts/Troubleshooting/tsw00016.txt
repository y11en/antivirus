// Функция поиска проблемы
function Check : dword;
var
 X : integer;
begin
 Result := 0;
 X := RegKeyIntParamRead('HKCU', 'Software\Microsoft\Windows\CurrentVersion\Internet Settings\Zones\3\', '1201');
 if X <> 3 then
  Result := 3;
end;

// Функция исправления проблемы
function Fix : dword;
begin
 RegKeyIntParamWrite('HKCU','Software\Microsoft\Windows\CurrentVersion\Internet Settings\Zones\3\', '1201', 3);
 Result := 0;
end;
