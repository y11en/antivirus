// $AVZ1532
//$ Отключено отображение значков в трее
// Функция поиска проблемы                                                                          
function Check : dword;
begin
 Result := 0;
 // Блокировка
 if RegKeyIntParamRead('HKCU', 'Software\Microsoft\Windows\CurrentVersion\Policies\Explorer', 'NoTrayItemsDisplay') > 0 then begin
   Result := 3;
   exit;
  end;
end;

// Функция исправления проблемы                                                                          
function Fix : dword;
begin
 RegKeyParamDel('HKCU', 'Software\Microsoft\Windows\CurrentVersion\Policies\Explorer', 'NoTrayItemsDisplay');
 Result := 0;
end;
